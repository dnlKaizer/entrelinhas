import { Flex, Typography, Button } from 'antd';
import { PlusOutlined } from '@ant-design/icons';
import { useNavigate } from 'react-router-dom';

// Importe o BookCard se estiver em outro arquivo
// import BookCard from './BookCard'; 

const { Text, Title } = Typography;

interface BookCategoryProps {
  title: string;
  backgroundColor: string;
  books: any[];
  emptyText: string;
}

const BookCategory = ({ title, backgroundColor, books, emptyText }: BookCategoryProps) => {
  const navigate = useNavigate();
  const hasBooks = books && books.length > 0;

  return (
    <div style={{ 
      backgroundColor, 
      padding: '16px', 
      borderRadius: '16px', // Arredondamento maior como na sua imagem
      marginBottom: '20px',
      display: 'flex',
      flexDirection: 'column'
    }}>
      {/* Título Centralizado */}
      <Title level={5} style={{ marginTop: 0, textAlign: 'center' }}>{title}</Title>

      {hasBooks ? (
        <>
          {/* Grid de Livros */}
          <Flex gap="large" wrap="wrap" justify="center" style={{ padding: '10px 0' }}>
             {/* Substitua pelo seu componente BookCard */}
             {books.map((book) => (
               <div key={book.id}>
                  <img src={book.cover} alt={book.title} style={{ width: 90, borderRadius: 8 }} />
               </div>
             ))}
          </Flex>

          {/* NOVO: Rodapé com Contador e Botão no canto inferior direito */}
          <Flex justify="flex-end" align="center" gap="middle" style={{ marginTop: 'auto', paddingTop: '10px' }}>
            <Text type="secondary" style={{ fontSize: '12px', fontWeight: 500 }}>
              {books.length} {books.length === 1 ? 'livro' : 'livros'}
            </Text>
            
            <Button 
              type="primary" 
              shape="circle" 
              icon={<PlusOutlined />} 
              onClick={() => navigate('/add-book')}
              style={{ boxShadow: '0 2px 6px rgba(0,0,0,0.15)' }} // Dá um leve destaque ao botão
            />
          </Flex>
        </>
      ) : (
        /* ESTADO VAZIO (Mantém o botão grande no centro) */
        <Flex 
          align="center" 
          justify="center"
          direction="column"
          gap="middle"
          onClick={() => navigate('/add-book')} 
          style={{ 
            cursor: 'pointer',
            padding: '30px',
            borderRadius: '12px',
            backgroundColor: '#ffffff',
            margin: '10px 0'
          }}
        >
          <Button shape="circle" size="large" icon={<PlusOutlined />} />
          <Text type="secondary">{emptyText}</Text>
        </Flex>
      )}
    </div>
  );
};

export default BookCategory;
